\section{Canal Rice}

\frame{
	\frametitle{Desvanecimento Rice}
\begin{block}{Modelagem Matemática}
A componente especular pode ser modelada matematicamente
como uma  das cópias do sinal que incidem sobre o móvel
\begin{equation}
r_0 = Re\{T_0 (t) e^{jw_ct}\} = A_0 \cos(w_ct + w_0t + \phi_0) .
\label{Van.3.34}
\end{equation}

  Como o sinal recebido é formado pela soma das componentes
decorrentes do espalhamento na vizinhança do móvel com a
componente especular, os coeficientes de transmissão podem ser
modelados por

\begin{equation}
u(t)= A_0 e^{j(w_0t + \phi_0)} + \sum_{n=1}^{M} A_n e^{j(w_nt + \phi_n)} .
\label{vand.3.36}
\end{equation}
\end{block}
}

\frame{
	\frametitle{Desvanecimento Rice}
\begin{block}{Modelagem Matemática}
O valor médio quadrático de $u(t)$ é dado por
\begin{equation}
b_R = A_0^2 + 2 \sigma^2 . \nonumber 
\label{vand.3.37}
\end{equation}

A potência da componente especular é determinada por $A_0^2$ e a potência das componentes espalhadas é dada por $2\sigma^2$. A razão entre ambas é o fator $K$ ou fator de Rice

\begin{equation}
K =  \dfrac{A_0^2}{2 \sigma^2}. \nonumber 
%\label{vand.3.38}
\end{equation}
\end{block}
}

\frame{
	\frametitle{Desvanecimento Rice}
\begin{block}{A fdp da distribuição de Rice para $r \geq 0$, é dada por}

\begin{center}
\begin{eqnarray}
f_{|u|}(r) = \dfrac{2r(K + 1)}{b_R}\mbox{exp}\left(-K - \dfrac{r^2(K + 1)}{b_R}\right) I_0\left(2r\sqrt{\dfrac{K(K +1)}{b_R}}\right),  \nonumber 
%\label{vand.3.44}
\end{eqnarray}
\end{center}
\end{block}

\begin{figure}[ht!]
    \centering
    \includegraphics[scale=0.3]{rice/fuc_dens_rice}
    %\caption{Canal com desvanecimento Rayleigh.}
    \label{fig:funcao}
\end{figure}

}

\frame{
	\frametitle{Estimação do fator $K$}
	\begin{block}{}
A partir de fatores e do modelo estatístico de primeira ordem para $K$, é possível chegar à seguinte equação
\begin{equation}
K \cong F_s F_h F_b K_o d^{\gamma} (d~\mbox{em~km}) . \nonumber
\label{eq:10}
\end{equation}
\vspace{-0.5cm}
\begin{enumerate}
\item $F_s$ é o fator da temporada

\begin{equation}
F_s = \left \{
\begin{array}{ll}
1.0, & \mbox{Verão~~(folhas)} ,\\
2.5, & \mbox{Inverno~~(sem~folhas)}  \nonumber \\	
\end{array}
\right.
\label{eq:11}
\end{equation}

\item $F_h$ é o fator da altura da antena

\begin{equation}
F_h = (h/3)^{0.46}~~(h~\mbox{em~metros}) . \nonumber
\label{eq:12}
\end{equation}
\item $F_b$ é o fator da largura do feixe da antena

\begin{equation}
F_b = (b/17)^{-0.62}~~(b~\mbox{em~graus}~(°)) . \nonumber
\label{eq:13}
\end{equation}
%$K_o$ e $\gamma$ são contantes de otimização encontradas por regressão~\cite{IEEE802}.

\end{enumerate}
\end{block}
}

\section{Conclusões}

\frame{
	\frametitle{Conclusões}
	\begin{block}{4-QAM}
	\begin{itemize}
	\item os melhores resultados foram 27°, 28° e 29°,
	\item ganho de 11 dB, para $\theta = 0°$,
	\item Não há diferença na BER.
\end{itemize}
	\end{block}
	\pause
	
		\begin{block}{16-QAM}
	\begin{itemize}
	\item os melhores resultados foram para os ângulos 16,8°, 21° e \textcolor{red}{ 31°},
	\item ganho de 8 dB, para $\theta = 0°$,
	\item Só existe diferença na BER para SNR $>$ 10 dB.
\end{itemize}
	\end{block}
	\pause
		\begin{block}{64-QAM}
		\begin{itemize}
		\item os melhores resultados obtidos foram 16,8°, 21° e \textcolor{red}{ 31°},
		\item ganho de 6 dB, para $\theta = 0°$,
		\item Só existe diferença na BER para SNR $>$ 15 dB.
		\end{itemize}
		\end{block}
}
